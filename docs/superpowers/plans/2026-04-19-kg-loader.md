# KG Loader + Bridges YAML Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `data/kg_bridges.yaml` seed data, a DB-backed async loader (`src/kg/loader.py`), read-only ORM models for slug queries, and integration tests with an in-transaction seed fixture.

**Architecture:** `loader.py` owns one public function (`load_sources`) that queries published topics/KCs/questions from Postgres, parses `kg_bridges.yaml` via PyYAML + private Pydantic models, validates referential integrity, then returns an immutable `LoadedSources`. Bridges YAML uses human-friendly string IDs (`CON-*`); the loader transforms them into proper `KGEdge`/`KGConcept` objects. `models_readonly.py` maps the `slug`/`status`/`review_status` columns that were added in the schema_v1 migration but are absent from `src/models/content.py`.

**Tech Stack:** Python 3.11, SQLAlchemy 2.0 async, Pydantic v2, PyYAML 6.x, pytest-asyncio, PostgreSQL 15

```python
"""Read-only ORM mappings for Knowledge Graph queries.

Uses a separate DeclarativeBase (KGReadBase) to avoid conflicts with
src.models.base.Base. These models are SELECT-only — never INSERT/UPDATE/DELETE.

The slug, status, and review_status columns were added in the schema_v1 Alembic
migration (20260418_schema_v1) but are absent from src/models/content.py.
"""

import uuid

from sqlalchemy import String
from sqlalchemy.dialects.postgresql import UUID
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column


class KGReadBase(DeclarativeBase):
    """Separate declarative base for KG read-only ORM models."""


class TopicRO(KGReadBase):
    """Read-only mapping to the topics table.

    Args:
        id: Topic primary key.
        slug: URL-safe unique identifier (nullable for legacy pre-migration rows).
        name: Display name.
        module_id: Parent module FK (UUID stored as-is).
        status: Content lifecycle status — 'draft' | 'published' | 'archived'.
    """

    __tablename__ = "topics"

    id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), primary_key=True)
    slug: Mapped[str | None] = mapped_column(String(100), nullable=True)
    name: Mapped[str] = mapped_column(String(255), nullable=False)
    module_id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), nullable=False)
    status: Mapped[str | None] = mapped_column(String(50), nullable=True)


class KnowledgeComponentRO(KGReadBase):
    """Read-only mapping to the knowledge_components table.

    Args:
        id: KC primary key.
        slug: URL-safe unique identifier (nullable for legacy rows).
        name: Display name.
        topic_id: Parent topic FK.
    """

    __tablename__ = "knowledge_components"

    id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), primary_key=True)
    slug: Mapped[str | None] = mapped_column(String(100), nullable=True)
    name: Mapped[str] = mapped_column(String(255), nullable=False)
    topic_id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), nullable=False)


class QuestionRO(KGReadBase):
    """Read-only mapping to the questions table.

    Args:
        id: Question primary key.
        item_id: Human-readable identifier (e.g. 'ITEM-001-00001').
        topic_id: Parent topic FK.
        review_status: Editorial status — 'draft' | 'published' | 'retired'.
    """

    __tablename__ = "questions"

    id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), primary_key=True)
    item_id: Mapped[str] = mapped_column(String(50), nullable=False)
    topic_id: Mapped[uuid.UUID] = mapped_column(UUID(as_uuid=True), nullable=False)
    review_status: Mapped[str | None] = mapped_column(String(50), nullable=True)
```

```python
"""Knowledge Graph data loader.

Loads published curriculum data from the database and bridges YAML,
validates referential integrity, and returns an immutable LoadedSources.

Public API: load_sources(session, data_dir) -> LoadedSources
"""

from __future__ import annotations

import uuid
from pathlib import Path
from typing import Any

import yaml
from pydantic import BaseModel
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from src.kg.models_readonly import KnowledgeComponentRO, QuestionRO, TopicRO
from src.kg.schemas import BridgesDoc, KCRef, KGConcept, KGEdge, LoadedSources, TopicRef

# Deterministic UUID namespace for concept string IDs (e.g. "CON-self-attention").
# Using a fixed namespace ensures idempotent UUID generation across build_kg runs.
_KG_NAMESPACE = uuid.UUID("20260419-beef-cafe-0000-000000000000")


# ---------------------------------------------------------------------------
# Private schemas for YAML parsing (YAML uses human IDs, not UUIDs)
# ---------------------------------------------------------------------------


class _ConceptEntry(BaseModel):
    """YAML concept entry — uses string ID like 'CON-self-attention'.

    Args:
        id: Human-readable concept identifier.
        name: Display name.
        description: Optional explanation.
        canonical_kc_slug: Authoritative KC slug for this concept.
    """

    id: str
    name: str
    description: str | None = None
    canonical_kc_slug: str | None = None


class _InstanceOfEntry(BaseModel):
    """YAML instance_of entry linking a KC to a concept.

    Args:
        kc_slug: Slug of the KnowledgeComponent.
        concept_id: String ID of the target concept.
    """

    kc_slug: str
    concept_id: str


class _TransfersToEntry(BaseModel):
    """YAML transfers_to entry representing topic-level knowledge transfer.

    Args:
        from_topic: Source topic slug.
        to_topic: Destination topic slug.
        weight: Transfer strength in [0.0, 1.0].
        reason: Human-readable justification.
    """

    from_topic: str
    to_topic: str
    weight: float
    reason: str = ""


class _BridgesYAML(BaseModel):
    """Internal schema for parsing the full kg_bridges.yaml file.

    Args:
        concepts: Concept node definitions.
        instance_of: KC→Concept membership edges.
        transfers_to: Topic→Topic transfer edges.
    """

    concepts: list[_ConceptEntry] = []
    instance_of: list[_InstanceOfEntry] = []
    transfers_to: list[_TransfersToEntry] = []
```

```python
def _parse_bridges_yaml(yaml_path: Path) -> _BridgesYAML:
    """Parse kg_bridges.yaml and validate its structure.

    Args:
        yaml_path: Absolute path to the YAML file.

    Returns:
        Validated _BridgesYAML instance.

    Raises:
        yaml.YAMLError: If the file contains invalid YAML syntax.
        pydantic.ValidationError: If the YAML structure does not match the schema.
    """
    with yaml_path.open("r", encoding="utf-8") as f:
        raw: dict[str, Any] = yaml.safe_load(f) or {}
    return _BridgesYAML.model_validate(raw)
```

```python
def _build_bridges_doc(raw: _BridgesYAML) -> BridgesDoc:
    """Transform parsed YAML into a validated BridgesDoc.

    Converts human-readable concept string IDs to deterministic UUIDs via
    uuid.uuid5(_KG_NAMESPACE, concept_id_string).

    Args:
        raw: Parsed YAML data from _parse_bridges_yaml.

    Returns:
        Immutable BridgesDoc with KGConcept and KGEdge objects.
    """
    concepts = [
        KGConcept(
            id=uuid.uuid5(_KG_NAMESPACE, entry.id),
            name=entry.name,
            description=entry.description,
            canonical_kc_slug=entry.canonical_kc_slug,
            source="manual",
        )
        for entry in raw.concepts
    ]

    instance_of = [
        KGEdge(
            src_kind="concept",
            src_ref=entry.concept_id,
            dst_kind="kc",
            dst_ref=entry.kc_slug,
            type="INSTANCE_OF",
            weight=1.0,
            source="schema",
        )
        for entry in raw.instance_of
    ]

    transfers_to = [
        KGEdge(
            src_kind="topic",
            src_ref=entry.from_topic,
            dst_kind="topic",
            dst_ref=entry.to_topic,
            type="TRANSFERS_TO",
            weight=entry.weight,
            source="schema",
            meta={"reason": entry.reason} if entry.reason else None,
        )
        for entry in raw.transfers_to
    ]

    return BridgesDoc(concepts=concepts, instance_of=instance_of, transfers_to=transfers_to)
```

```python
async def _query_topics(session: AsyncSession) -> list[TopicRO]:
    """Fetch all published topics that have a slug from the database.

    Args:
        session: Active async database session.

    Returns:
        List of TopicRO rows with non-null slug and status='published'.
    """
    result = await session.execute(
        select(TopicRO)
        .where(TopicRO.status == "published")
        .where(TopicRO.slug.isnot(None))
    )
    return list(result.scalars().all())
```

```python
async def _query_kcs(
    session: AsyncSession,
    topic_map: dict[uuid.UUID, TopicRef],
) -> list[KCRef]:
    """Fetch all knowledge components with a slug, enriched with parent topic info.

    Args:
        session: Active async database session.
        topic_map: Mapping from topic UUID to TopicRef (built from _query_topics output).

    Returns:
        List of KCRef with topic_slug and module_slug resolved via topic_map.
        KCs whose parent topic is not in topic_map are skipped.
    """
    result = await session.execute(
        select(KnowledgeComponentRO).where(KnowledgeComponentRO.slug.isnot(None))
    )
    rows = list(result.scalars().all())

    kcs: list[KCRef] = []
    for row in rows:
        topic = topic_map.get(row.topic_id)
        if topic is None:
            continue  # KC belongs to an unpublished or slug-less topic — skip
        kcs.append(
            KCRef(
                slug=row.slug,  # type: ignore[arg-type]
                topic_slug=topic.slug,
                module_slug=topic.module_slug,
                name=row.name,
            )
        )
    return kcs
```

```python
def _check_integrity(
    topics: list[TopicRef],
    kcs: list[KCRef],
    raw: _BridgesYAML,
) -> None:
    """Validate all cross-references in the parsed YAML against loaded DB data.

    Args:
        topics: Published topics from DB.
        kcs: Knowledge components from DB.
        raw: Parsed YAML bridges data.

    Raises:
        ValueError: On first referential integrity violation, with a message
            identifying the offending field and value.
    """
    topic_slug_set = {t.slug for t in topics}
    kc_slug_set = {kc.slug for kc in kcs}
    concept_id_set = {c.id for c in raw.concepts}

    for entry in raw.instance_of:
        if entry.kc_slug not in kc_slug_set:
            raise ValueError(
                f"instance_of references unknown kc_slug: '{entry.kc_slug}'"
            )
        if entry.concept_id not in concept_id_set:
            raise ValueError(
                f"instance_of references unknown concept_id: '{entry.concept_id}'"
            )

    for entry in raw.transfers_to:
        if entry.from_topic not in topic_slug_set:
            raise ValueError(
                f"transfers_to references unknown topic: '{entry.from_topic}'"
            )
        if entry.to_topic not in topic_slug_set:
            raise ValueError(
                f"transfers_to references unknown topic: '{entry.to_topic}'"
            )

    for concept in raw.concepts:
        if concept.canonical_kc_slug and concept.canonical_kc_slug not in kc_slug_set:
            raise ValueError(
                f"concept '{concept.id}' canonical_kc_slug "
                f"'{concept.canonical_kc_slug}' not found in knowledge_components"
            )
```

```python
async def load_sources(session: AsyncSession, data_dir: Path) -> LoadedSources:
    """Load curriculum data from DB and bridges YAML with integrity validation.

    Queries published topics, KCs, and questions from the database, parses
    data_dir/kg_bridges.yaml, validates all cross-references, and returns
    an immutable LoadedSources.

    Args:
        session: Active async database session.
        data_dir: Directory containing kg_bridges.yaml.

    Returns:
        Validated LoadedSources with topics, KCs, questions, and bridges.

    Raises:
        yaml.YAMLError: If kg_bridges.yaml has invalid YAML syntax.
        pydantic.ValidationError: If the YAML structure is invalid.
        ValueError: If any cross-reference in the YAML is missing from the DB.
    """
    topic_rows = await _query_topics(session)

    topics: list[TopicRef] = [
        TopicRef(
            slug=row.slug,  # type: ignore[arg-type]
            module_slug=str(row.module_id),  # Phase 0: UUID string; Phase 1: join modules
            name=row.name,
            prerequisite_topic_slugs=[],
        )
        for row in topic_rows
    ]

    topic_map: dict[uuid.UUID, TopicRef] = {
        row.id: ref for row, ref in zip(topic_rows, topics)
    }

    kcs = await _query_kcs(session, topic_map)
    questions = await _query_questions(session)

    raw = _parse_bridges_yaml(data_dir / "kg_bridges.yaml")
    _check_integrity(topics, kcs, raw)
    bridges = _build_bridges_doc(raw)

    return LoadedSources(topics=topics, kcs=kcs, questions=questions, bridges=bridges)
```
